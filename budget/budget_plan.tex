CategoryItem {
    self:
        amount
        subject
        date

    def __init__(self, amount, subject, date):
            self.amount = amount
            self.subject = subject
            self.date = date
            


    __str__

    def __add_(self, other):
        if isinstance(other, CategoryItem):
            return CategoryItem(self.amount + other.amount)
        return NotImplemented

}
